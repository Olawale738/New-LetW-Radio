"""
Desktop Automation Module
Handles taking control of desktop, opening software, and capturing output
"""

import os
import time
import logging
import subprocess
from pathlib import Path
from typing import Optional, Dict, List
import pyautogui
import pygetwindow as gw
from PIL import Image, ImageGrab
import pywinauto
from pywinauto import Application, timings

from config import SOFTWARE_PATHS, ANALYSIS_SETTINGS, OUTPUT_DIR, LOGS_DIR

# Configure logging
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s [%(levelname)s] %(name)s: %(message)s',
    handlers=[
        logging.FileHandler(LOGS_DIR / 'desktop_automation.log'),
        logging.StreamHandler()
    ]
)
logger = logging.getLogger(__name__)

# Disable pyautogui failsafe
pyautogui.FAILSAFE = False
pyautogui.PAUSE = 0.5


class DesktopAutomation:
    """Handles desktop automation for statistical software control"""
    
    def __init__(self):
        self.active_window = None
        self.screenshot_delay = ANALYSIS_SETTINGS["screenshot_delay"]
        self.output_wait_time = ANALYSIS_SETTINGS["output_wait_time"]
        self.current_process = None
        
    def take_screenshot(self, region: Optional[tuple] = None, filename: str = "screenshot.png") -> Path:
        """
        Take a screenshot of the entire screen or a specific region
        
        Args:
            region: (left, top, width, height) or None for full screen
            filename: Name for the screenshot file
            
        Returns:
            Path to the saved screenshot
        """
        try:
            if region:
                screenshot = ImageGrab.grab(bbox=region)
            else:
                screenshot = ImageGrab.grab()
            
            output_path = OUTPUT_DIR / filename
            screenshot.save(output_path, quality=ANALYSIS_SETTINGS["screenshot_quality"])
            logger.info(f"Screenshot saved to: {output_path}")
            return output_path
        except Exception as e:
            logger.error(f"Failed to take screenshot: {e}")
            raise
    
    def capture_active_window(self, filename: str = "window_capture.png") -> Path:
        """Capture the currently active window"""
        try:
            if self.active_window:
                # Get window bounds
                left, top, width, height = self.active_window.left, self.active_window.top, \
                                           self.active_window.width, self.active_window.height
                
                # Capture the window region
                screenshot = ImageGrab.grab(bbox=(left, top, left + width, top + height))
                
                output_path = OUTPUT_DIR / filename
                screenshot.save(output_path, quality=ANALYSIS_SETTINGS["screenshot_quality"])
                logger.info(f"Window capture saved to: {output_path}")
                return output_path
            else:
                logger.warning("No active window set. Taking full screenshot instead.")
                return self.take_screenshot(filename=filename)
        except Exception as e:
            logger.error(f"Failed to capture window: {e}")
            raise
    
    def open_software(self, software_name: str, wait_for_load: int = 10) -> bool:
        """
        Open a statistical software application
        
        Args:
            software_name: Name of the software (stata, spss, nvivo, amos, r)
            wait_for_load: Seconds to wait for software to load
            
        Returns:
            True if successful, False otherwise
        """
        software_path = SOFTWARE_PATHS.get(software_name.lower())
        
        if not software_path or not os.path.exists(software_path):
            logger.warning(f"Software path not found for: {software_name}")
            logger.info(f"Attempting to launch {software_name} from Start menu...")
            return self._launch_from_start_menu(software_name)
        
        try:
            logger.info(f"Opening {software_name} from: {software_path}")
            self.current_process = subprocess.Popen(
                software_path,
                stdout=subprocess.PIPE,
                stderr=subprocess.PIPE
            )
            
            # Wait for software to load
            time.sleep(wait_for_load)

    def filter_data(self, conditions: Dict) -> pd.DataFrame:
        """
        Filter data based on conditions
        
        Args:
            conditions: Dictionary of {column: value} or {column: (operator, value)}
            
        Returns:
            Filtered DataFrame
        """
        if self.data is None:
            raise ValueError("No data loaded")
        
        filtered = self.data.copy()
        
        for col, condition in conditions.items():
            if col not in self.data.columns:
                continue
                
            if isinstance(condition, tuple):
                op, val = condition
                if op == '==':
                    filtered = filtered[filtered[col] == val]
                elif op == '!=':
                    filtered = filtered[filtered[col] != val]
                elif op == '>':
                    filtered = filtered[filtered[col] > val]
                elif op == '<':
                    filtered = filtered[filtered[col] < val]
                elif op == '>=':
                    filtered = filtered[filtered[col] >= val]
                elif op == '<=':
                    filtered = filtered[filtered[col] <= val]
                elif op == 'in':
                    filtered = filtered[filtered[col].isin(val)]
            else:
                filtered = filtered[filtered[col] == condition]
        
        logger.info(f"Filtered to {len(filtered)} rows")
        return filtered
    
    def handle_missing(self, method: str = 'drop', 
                      fill_value: any = None,
                      columns: List[str] = None) -> pd.DataFrame:
        """
        Handle missing values
        
        Args:
            method: 'drop', 'fill', 'mean', 'median', 'mode'
            fill_value: Value to fill with (if method='fill')
            columns: Columns to process (None = all)
            
        Returns:
            DataFrame with handled missing values
        """
        if self.data is None:
            raise ValueError("No data loaded")
        
        df = self.data.copy()
        cols = columns or df.columns
        
        if method == 'drop':
            df = df.dropna(subset=cols)
        elif method == 'fill' and fill_value is not None:
            df[cols] = df[cols].fillna(fill_value)
        elif method == 'mean':
            for col in cols:
                if pd.api.types.is_numeric_dtype(df[col]):
                    df[col] = df[col].fillna(df[col].mean())
        elif method == 'median':
            for col in cols:
                if pd.api.types.is_numeric_dtype(df[col]):
                    df[col] = df[col].fillna(df[col].median())
        elif method == 'mode':
            for col in cols:
                df[col] = df[col].fillna(df[col].mode()[0] if len(df[col].mode()) > 0 else None)
        
        self.data = df
        logger.info(f"Handled missing values using method: {method}")
        
        return df

class PythonStatistics:
    """Python-based statistical analysis"""
    
    def __init__(self):
        self.results = {}
    
    def descriptive_stats(self, df, variables=None):
        if variables is None:
            variables = df.select_dtypes(include=[np.number]).columns.tolist()
        
        desc = df[variables].describe()
        return {
            'descriptives': desc.to_dict(),
            'n': len(df)
        }
    
    def correlation(self, df, variables, method='pearson'):
        data = df[variables].dropna()
        corr_matrix = data.corr(method=method)
        
        p_values = pd.DataFrame(np.zeros((len(variables), len(variables))),
                               index=variables, columns=variables)
        
        for i, var1 in enumerate(variables):
            for j, var2 in enumerate(variables):
                if i != j:
                    if method == 'spearman':
                        _, p = spearmanr(data[var1], data[var2])
                    elif method == 'kendall':
                        _, p = kendalltau(data[var1], data[var2])
                    else:
                        _, p = stats.pearsonr(data[var1], data[var2])
                    p_values.loc[var1, var2] = p
        
        return {
            'method': method,
            'correlation_matrix': corr_matrix.to_dict(),
            'p_values': p_values.to_dict(),
            'n': len(data)
        }
    
    def ols_regression(self, df, dependent, independents):
        y = df[dependent].dropna()
        X = df[independents].dropna()
        
        common_idx = y.index.intersection(X.index)
        y = y.loc[common_idx]
        X = X.loc[common_idx]
        
        X = sm.add_constant(X)
        model = sm.OLS(y, X).fit()
        
        return {
            'model': 'OLS',
            'dependent': dependent,
            'n': len(y),
            'r_squared': float(model.rsquared),
            'adj_r_squared': float(model.rsquared_adj),
            'f_statistic': float(model.fvalue),
            'f_pvalue': float(model.f_pvalue),
            'coefficients': {var: float(model.params[var]) for var in model.params.index},
            'std_errors': {var: float(model.bse[var]) for var in model.bse.index},
            'p_values': {var: float(model.pvalues[var]) for var in model.pvalues.index}
        }
    
    def ttest_independent(self, df, group_var, test_var):
        groups = df[group_var].unique()
        if len(groups) != 2:
            return {'error': 'Need exactly 2 groups'}
        
        group1 = df[df[group_var] == groups[0]][test_var].dropna()
        group2 = df[df[group_var] == groups[1]][test_var].dropna()
        
        t_stat, p_value = stats.ttest_ind(group1, group2)
        
        return {
            'test': 'T-Test',
            'group_var': group_var,
            'test_var': test_var,
            't_statistic': float(t_stat),
            'p_value': float(p_value),
            'group1_mean': float(group1.mean()),
            'group2_mean': float(group2.mean())
        }
    
    def one_way_anova(self, df, dependent, factor):
        groups = df[factor].unique()
        group_data = [df[df[factor] == g][dependent].dropna() for g in groups]
        
        f_stat, p_value = stats.f_oneway(*group_data)
        
        return {
            'test': 'ANOVA',
            'dependent': dependent,
            'factor': factor,
            'f_statistic': float(f_stat),
            'p_value': float(p_value)
        }
    
    def mediation_analysis(self, df, dv, iv, mediator):
        # Path a
        X1 = sm.add_constant(df[[iv]])
        path_a = sm.OLS(df[mediator], X1).fit()
        
        # Path b and c'
        X2 = sm.add_constant(df[[iv, mediator]])
        path_bc = sm.OLS(df[dv], X2).fit()
        
        return {
            'test': 'Mediation',
            'path_a': float(path_a.params[iv]),
            'path_b': float(path_bc.params[mediator]),
            'direct_effect': float(path_bc.params[mediator]),
            'indirect_effect': float(path_a.params[iv] * path_bc.params[mediator]),
            'total_effect': float(path_bc.params[iv])
        }

                        _, p = spearmanr(data[var1], data[var2])

class PythonStatistics:
    """Statistical analysis using Python"""
    
    def descriptive_stats(self, df, variables=None):
        if variables is None:
            variables = df.select_dtypes(include=[np.number]).columns.tolist()
        
        desc = df[variables].describe()
        return {
            'descriptives': desc.to_dict(),
            'n': len(df)
        }
    
    def correlation(self, df, variables, method='pearson'):
        data = df[variables].dropna()
        corr_matrix = data.corr(method=method)
        
        p_values = pd.DataFrame(np.zeros((len(variables), len(variables))),
                               index=variables, columns=variables)
        
        for i, var1 in enumerate(variables):
            for j, var2 in enumerate(variables):
                if i != j:
                    if method == 'spearman':
                        _, p = spearmanr(data[var1], data[var2])
                    else:
                        _, p = stats.pearsonr(data[var1], data[var2])
                    p_values.loc[var1, var2] = p
        
        return {
            'method': method,
            'correlation_matrix': corr_matrix.to_dict(),
            'p_values': p_values.to_dict(),
            'n': len(data)
        }
    
    def ols_regression(self, df, dependent, independents):
        y = df[dependent].dropna()
        X = df[independents].dropna()
        
        common_idx = y.index.intersection(X.index)
        y = y.loc[common_idx]
        X = X.loc[common_idx]
        
        X = sm.add_constant(X)
        model = sm.OLS(y, X).fit()
        
        return {
            'model': 'OLS',
            'dependent': dependent,
            'n': len(y),
            'r_squared': float(model.rsquared),
            'adj_r_squared': float(model.rsquared_adj),
            'f_statistic': float(model.fvalue),
            'f_pvalue': float(model.f_pvalue),
            'coefficients': {var: float(model.params[var]) for var in model.params.index},
            'std_errors': {var: float(model.bse[var]) for var in model.bse.index},
            'p_values': {var: float(model.pvalues[var]) for var in model.pvalues.index},
            'conf_int_lower': {var: float(model.conf_int()[0][var]) for var in model.conf_int().index},
            'conf_int_upper': {var: float(model.conf_int()[1][var]) for var in model.conf_int().index},
            't_values': {var: float(model.tvalues[var]) for var in model.tvalues.index}
        }
    
    def ttest_independent(self, df, group_var, test_var):
        try:
            groups = df[group_var].unique()
            if len(groups) != 2:
                return {'error': 'Need exactly 2 groups', 'skip': True}
            
            group1 = df[df[group_var] == groups[0]][test_var].dropna()
            group2 = df[df[group_var] == groups[1]][test_var].dropna()
            
            if len(group1) < 2 or len(group2) < 2:
                return {'error': 'Insufficient data', 'skip': True}
            
            t_stat, p_value = stats.ttest_ind(group1, group2)
            
            # Calculate effect size (Cohen's d)
            pooled_std = np.sqrt(((len(group1)-1)*group1.std()**2 + (len(group2)-1)*group2.std()**2) / (len(group1)+len(group2)-2))
            cohens_d = (group1.mean() - group2.mean()) / pooled_std if pooled_std > 0 else 0
            
            return {
                'test': 'T-Test',
                'group_var': group_var,
                'test_var': test_var,
                't_statistic': float(t_stat),
                'p_value': float(p_value),
                'cohens_d': float(cohens_d),
                'group1_mean': float(group1.mean()),
                'group2_mean': float(group2.mean()),
                'group1_std': float(group1.std()),
                'group2_std': float(group2.std()),
                'group1_n': len(group1),
                'group2_n': len(group2)
            }
        except Exception as e:
            return {'error': str(e), 'skip': True}
    
    def one_way_anova(self, df, dependent, factor):
        groups = df[factor].unique()
        group_data = [df[df[factor] == g][dependent].dropna() for g in groups]
        
        f_stat, p_value = stats.f_oneway(*group_data)
        
        # Calculate effect size (eta-squared)
        all_data = df[dependent].dropna()
        ss_between = sum(len(g) * (g.mean() - all_data.mean())**2 for g in group_data)
        ss_total = sum((all_data - all_data.mean())**2)
        eta_squared = ss_between / ss_total if ss_total > 0 else 0
        
        return {
            'test': 'ANOVA',
            'dependent': dependent,
            'factor': factor,
            'f_statistic': float(f_stat),
            'p_value': float(p_value),
            'eta_squared': float(eta_squared),
            'n_groups': len(groups)
        }
    
    def mediation_analysis(self, df, dv, iv, mediator):

class APATableGenerator:
    """Generate APA style tables"""
    
    def create_regression_table(self, results):
        """Create APA regression table"""
        rows = []
        
        # Header
        rows.append(["Variable", "B", "SE", "β", "t", "p", "95% CI"])
        rows.append(["-" * 50] * 7)
        
        # Constant
        const_b = results['coefficients'].get('const', 0)
        const_se = results['std_errors'].get('const', 0)
        const_p = results['p_values'].get('const', 1)
        const_t = results['t_values'].get('const', 0)
        
        # Calculate standardized coefficient (beta)
        # For simplicity, we'll use unstandardized B
        rows.append(["(Constant)", f"{const_b:.2f}", f"{const_se:.2f}", "-", f"{const_t:.2f}", 
                    f"{const_p:.4f}" if const_p >= 0.001 else "<.001", "-"])
        
        # Predictors
        for var in results['coefficients']:
            if var == 'const':
                continue
            
            b = results['coefficients'][var]
            se = results['std_errors'][var]
            p = results['p_values'].get(var, 1)
            t = results['t_values'].get(var, 0)
            ci_low = results['conf_int_lower'].get(var, 0)
            ci_high = results['conf_int_upper'].get(var, 0)
            
            # Shorten variable name
            short_var = var.split('\n')[0][:25]
            
            rows.append([
                short_var,
                f"{b:.3f}",
                f"{se:.3f}",
                "-",  # β would need additional calculation
                f"{t:.2f}",
                f"{p:.4f}" if p >= 0.001 else "<.001",
                f"[{ci_low:.3f}, {ci_high:.3f}]"
            ])
        
        # Model fit
        rows.append("")    
        rows.append([f"R² = {results['r_squared']:.3f}, Adjusted R² = {results['adj_r_squared']:.3f}"])
        rows.append([f"F({len(results['coefficients'])-1}, {results['n']-len(results['coefficients'])}) = {results['f_statistic']:.2f}, p < .001" 
                    if results['f_pvalue'] < 0.001 else f"p = {results['f_pvalue']:.4f}"])
        
        return rows
    
    def create_correlation_table(self, results, variables):
        """Create APA correlation matrix"""
        rows = []
        
        # Header
        header = [""]
        for v in variables:
            header.append(v.split('\n')[0][:8])
        rows.append(header)
        
        # Correlation matrix
        corr = results['correlation_matrix']
        pvals = results['p_values']
        
        for i, var1 in enumerate(variables):
            row = [var1.split('\n')[0][:8]]
            for var2 in variables:
                r = corr.get(var1, {}).get(var2, 0)
                p = pvals.get(var1, {}).get(var2, 1)
                
                if var1 == var2:
                    row.append("1.00")
                else:
                    sig = "*" if p < 0.05 else ""
                    row.append(f"{r:.2f}{sig}")
            rows.append(row)
        
        rows.append(f"N = {results['n']}")
        rows.append("* p < .05")
        
        return rows


class WordDocumentGenerator:
    """Generate Word documents"""
    
    def __init__(self):
        self.doc = None
        self._init_doc()
    
    def _init_doc(self):
        """Initialize Word document"""
        try:
            from docx import Document
            from docx.shared import Inches, Pt, RGBColor
            from docx.enum.text import WD_ALIGN_PARAGRAPH
            self.doc = Document()
            self.has_docx = True
        except ImportError:
            self.has_docx = False
            print("Note: python-docx not installed. Installing...")
            os.system('pip install python-docx')
            try:
                from docx import Document
                self.doc = Document()
                self.has_docx = True
            except:
                self.has_docx = False
    
    def add_title(self, title):
        if not self.has_docx:
            return
        heading = self.doc.add_heading(title, 0)
        heading.alignment = WD_ALIGN_PARAGRAPH.CENTER
    
    def add_heading(self, text, level=1):
        if not self.has_docx:
            return
        self.doc.add_heading(text, level)
    
    def add_paragraph(self, text, bold=False):
        if not self.has_docx:
            return
        p = self.doc.add_paragraph(text)
        if bold:
            p.runs[0].bold = True
        return p
    
    def add_table(self, rows):
        if not self.has_docx:
            return
        table = self.doc.add_table(rows=len(rows), cols=len(rows[0]) if rows else 0)
        table.style = 'Table Grid'
        
        for i, row_data in enumerate(rows):
            row = table.rows[i]
            for j, cell_data in enumerate(row_data):
                if j < len(row.cells):
                    row.cells[j].text = str(cell_data)
    
    def save(self, path):
        if self.has_docx:
            self.doc.save(path)
            print(f"Word document saved: {path}")
            return path
        else:
            # Save as text
            with open(path.replace('.docx', '.txt'), 'w', encoding='utf-8') as f:
                f.write("Word document generation requires: pip install python-docx\n")
            return path.replace('.docx', '.txt')
