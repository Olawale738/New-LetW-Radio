================
Parses CV/Resume documents and extracts relevant information.
Supports DOCX and PDF formats.
"""

import os
import re
from datetime import datetime
from typing import Dict, List, Optional
import python_docx
from loguru import logger


class CVParser:
    """Parser for CV/Resume documents."""
    
    def __init__(self, cv_path: str):
        """Initialize CV parser with path to CV file."""
        self.cv_path = cv_path
        self.raw_text = ""
        self.parsed_data = {}
        
    def read_docx(self) -> str:
        """Read content from DOCX file."""
        try:
            doc = python_docx.Document(self.cv_path)
            text = []
            for para in doc.paragraphs:
                text.append(para.text)
            return "\n".join(text)
        except Exception as e:
            logger.error(f"Error reading DOCX file: {e}")
            return ""
    
    def read_pdf(self) -> str:
        """Read content from PDF file."""
        try:
            import PyPDF2
            with open(self.cv_path, 'rb') as file:
                reader = PyPDF2.PdfReader(file)
                text = []
                for page in reader.pages:
                    text.append(page.extract_text())
                return "\n".join(text)
        except Exception as e:
            logger.error(f"Error reading PDF file: {e}")
            return ""
    
    def load_cv(self) -> str:
        """Load CV content from file."""
        if not os.path.exists(self.cv_path):
            logger.error(f"CV file not found: {self.cv_path}")
            return ""
        
        ext = os.path.splitext(self.cv_path)[1].lower()
        
        if ext == '.docx':
            self.raw_text = self.read_docx()
        elif ext == '.pdf':
            self.raw_text = self.read_pdf()
        else:
            logger.error(f"Unsupported file format: {ext}")
            return ""
        
        logger.info(f"CV loaded successfully: {len(self.raw_text)} characters")
        return self.raw_text
    
    def extract_name(self) -> str:
        """Extract candidate name from CV."""
        lines = self.raw_text.split('\n')
        # Usually the name is in the first few lines
        for line in lines[:5]:
            line = line.strip()
            if line and not '@' in line and not '+' in line:
                # Skip lines that look like addresses or emails
                if not any(x in line.lower() for x in ['email', 'tel', 'phone', 'address', 'http']):
                    # Clean the name
                    name = re.sub(r'[^\w\s]', '', line).strip()
                    if len(name.split()) >= 2 and len(name) < 50:
                        return name
        return ""
    
    def extract_email(self) -> str:
        """Extract email address from CV."""
        email_pattern = r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b'
        match = re.search(email_pattern, self.raw_text)
        return match.group(0) if match else ""
    
    def extract_phone(self) -> str:
        """Extract phone number from CV."""
        # Multiple patterns for international phone numbers
        phone_patterns = [
            r'\+\d{1,3}[-\s]?\(?\d{1,4}\)?[-\s]?\d{1,4}[-\s]?\d{1,9}',
            r'\b\d{10,12}\b',
            r'\b0\d{4,5}[-\s]?\d{5,7}\b',
            r'\b\d{3}[-\s]?\d{3}[-\s]?\d{4}\b'
        ]
        
        for pattern in phone_patterns:
            match = re.search(pattern, self.raw_text)
            if match:
                return match.group(0)
        return ""
    
    def extract_address(self) -> str:
        """Extract address from CV."""
        # Look for UK-style addresses with postcodes
        uk_postcode_pattern = r'[A-Z]{1,2}\d{1,2}[A-Z]?\s?\d[A-Z]{2}'

        if match:

            # Clean up
            lines = address_text.split('\n')

                    return line.strip()
        
        return ""

        return ""

            return False

def run_single_cycle(platform: str = None, apply_only: bool = False):
    """Run a single application cycle."""
    logger.info("Initializing Auto Job Applying Bot...")
    
    # Initialize the auto-applier
    applier = AutoApplier()
    
    if not applier.initialize():
        logger.error("Failed to initialize the bot. Please check your configuration.")
        return False
    
    # Print CV profile summary
    cv_data = applier.cv_data
    if cv_data:
        logger.info(f"CV Profile: {cv_data.get('name', 'Unknown')}")
        logger.info(f"Email: {cv_data.get('email', 'Not found')}")
        logger.info(f"Skills: {', '.join(cv_data.get('skills', [])[:5])}...")
    
    if apply_only:
        # Only apply to existing jobs
        logger.info("Running in APPLY-ONLY mode...")
        count = applier.apply_to_jobs(platform=platform)
        logger.info(f"Applied to {count} jobs")
    else:
        # Full cycle: search and apply
        logger.info("Running full cycle: Search + Apply...")
        result = applier.run_cycle()
        
        if result.get('success'):
            logger.info(f"✅ Cycle completed successfully!")
            logger.info(f"   Jobs found: {result.get('jobs_found')}")
            logger.info(f"   Applications sent: {result.get('applications')}")
            logger.info(f"   Time elapsed: {result.get('elapsed_seconds'):.1f} seconds")
        else:
            logger.error(f"❌ Cycle failed: {result.get('error')}")
            return False
    
    return True


def run_continuous_mode(interval_minutes: int = None):
    """Run the bot continuously in the background."""
    import schedule
    
    interval = interval_minutes or config.JOB_SEARCH_INTERVAL
    
    logger.info(f"Starting CONTINUOUS mode (every {interval} minutes)...")
    
    # Initialize
    applier = AutoApplier()
    
    if not applier.initialize():
        logger.error("Failed to initialize the bot. Please check your configuration.")
        return
    
    # Run immediately on start
    logger.info("Running initial cycle...")
    applier.run_cycle()
    
    # Schedule periodic runs
    schedule.every(interval).minutes.do(applier.run_cycle)
    
    logger.info(f"Bot is now running in the background. Press Ctrl+C to stop.")
    
    # Keep running
    try:
        while True:
            schedule.run_pending()
            
            # Print status every hour
            stats = applier.get_statistics()
            logger.debug(f"Stats: {stats['totals']}")
            
            # Sleep for 1 minute
            import time
            time.sleep(60)
            
    except KeyboardInterrupt:
        logger.info("Bot stopped by user.")
        return

"""
Background Runner for Auto Job Applying Bot
============================================
Run the bot in the background using Windows Task Scheduler.

Usage:
    python run_bot.py install   - Install scheduled task
    python run_bot.py start     - Start the bot immediately
    python run_bot.py stop      - Stop the bot
    python run_bot.py status    - Check bot status
"""

import sys
import os
import subprocess
import time
import signal
from datetime import datetime

# Configuration
BOT_SCRIPT = "main.py"
BOT_DIR = os.path.dirname(os.path.abspath(__file__))
PID_FILE = os.path.join(BOT_DIR, "bot.pid")
LOG_FILE = os.path.join(BOT_DIR, "logs", "background.log")


def ensure_dir(path):
    """Ensure directory exists."""
    os.makedirs(os.path.dirname(path), exist_ok=True)


def log_message(message):
    """Log message to file."""
    ensure_dir(LOG_FILE)
    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    with open(LOG_FILE, "a") as f:
        f.write(f"[{timestamp}] {message}\n")


def get_pid():
    """Get bot PID if running."""
    if os.path.exists(PID_FILE):
        try:
            with open(PID_FILE, "r") as f:
                return int(f.read().strip())
        except:
            return None
    return None


def is_running():
    """Check if bot is running."""
    pid = get_pid()
    if pid:
        try:
            # Check if process exists
            os.kill(pid, 0)
            return True
        except OSError:
            # Process doesn't exist
            return False
    return False


def start_bot():
    """Start the bot in background."""
    if is_running():
        print("Bot is already running!")
        return False
    
    print("Starting Auto Job Applying Bot...")
    log_message("Starting bot...")
    
    try:
        # Start Python process in background
        if sys.platform == "win32":
            # Windows: use start /b or createhidden
            CREATE_NO_WINDOW = 0x08000000
            process = subprocess.Popen(
                [sys.executable, BOT_SCRIPT, "--continuous", "--headless"],
                cwd=BOT_DIR,
                stdout=subprocess.PIPE,
                stderr=subprocess.PIPE,
                creationflags=CREATE_NO_WINDOW
            )
        else:
            # Unix: use nohup
            process = subprocess.Popen(
                [sys.executable, BOT_SCRIPT, "--continuous", "--headless"],
                cwd=BOT_DIR,
                stdout=open(os.devnull, "w"),
                stderr=open(os.devnull, "w"),
                start_new_session=True
            )
        
        # Save PID
        with open(PID_FILE, "w") as f:
            f.write(str(process.pid))
        
        log_message(f"Bot started with PID: {process.pid}")
        print(f"Bot started successfully! (PID: {process.pid})")
        return True
        
    except Exception as e:
        log_message(f"Error starting bot: {e}")
        print(f"Error starting bot: {e}")
        return False


def stop_bot():
    """Stop the running bot."""
    pid = get_pid()
    
    if not pid:
        print("Bot is not running (no PID file)")
        return False
    
    if not is_running():
        print("Bot is not running (process not found)")
        # Clean up PID file
        if os.path.exists(PID_FILE):
            os.remove(PID_FILE)
        return False
    
    try:
        # Try graceful shutdown first
        os.kill(pid, signal.SIGTERM)
        print("Sending stop signal...")
        time.sleep(2)
        
        # Check if stopped
        if not is_running():
            if os.path.exists(PID_FILE):
                os.remove(PID_FILE)
            log_message("Bot stopped successfully")
            print("Bot stopped successfully!")
            return True
        
        # Force kill if still running
        os.kill(pid, signal.SIGKILL)
        time.sleep(1)
        
        if os.path.exists(PID_FILE):
            os.remove(PID_FILE)
        
        log_message("Bot force stopped")
        print("Bot force stopped!")
        return True
        
    except Exception as e:
        log_message(f"Error stopping bot: {e}")
        print(f"Error stopping bot: {e}")
        return False


def install_task():
    """Install Windows Task Scheduler task."""
    if sys.platform != "win32":
        print("Task Scheduler is only available on Windows")
        return False
    
    task_name = "AutoJobBot"
    python_path = sys.executable
    script_path = os.path.join(BOT_DIR, BOT_SCRIPT)
    
    # Create task command
    cmd = [
        "schtasks",
        "/create",
        "/tn", task_name,
        "/tr", f'"{python_path}" "{script_path}" --continuous --headless',
        "/sc", "minute",
        "/mo", "30",  # Run every 30 minutes
        "/f"  # Force overwrite
    ]
    
    try:
        subprocess.run(cmd, check=True)
        log_message("Scheduled task installed")
        print(f"Scheduled task '{task_name}' installed successfully!")
        print(f"The bot will run every 30 minutes.")
        return True
    except Exception as e:
        log_message(f"Error installing task: {e}")
        print(f"Error installing task: {e}")
        return False


def remove_task():
    """Remove Windows Task Scheduler task."""
    if sys.platform != "win32":
        print("Task Scheduler is only available on Windows")
        return False
    
    task_name = "AutoJobBot"
    
    try:
        subprocess.run(["schtasks", "/delete", "/tn", task_name, "/f"], check=True)
        log_message("Scheduled task removed")
        print(f"Scheduled task '{task_name}' removed!")
        return True
    except Exception as e:
        print(f"Error removing task: {e}")
        return False


def check_status():
    """Check bot status."""
    if is_running():
        pid = get_pid()
        print(f"✅ Bot is RUNNING (PID: {pid})")
        
        # Try to get more info
        try:

            process = psutil.Process(pid)
            print(f"   Started: {datetime.fromtimestamp(process.create_time())}")
            print(f"   CPU: {process.cpu_percent():.1f}%")
            print(f"   Memory: {process.memory_info().rss / 1024 / 1024:.1f} MB")
        except ImportError:
            pass
    else:
        print("❌ Bot is NOT running")
    
    # Show log excerpt
    if os.path.exists(LOG_FILE):
        print("\nRecent log entries:")
        with open(LOG_FILE, "r") as f:
            lines = f.readlines()
            for line in lines[-5:]:
                print(f"   {line.strip()}")


def main():

    if len(sys.argv) < 2:
        print(__doc__)

        start_bot()
    elif command == "stop":
        stop_bot()
    elif command == "restart":
        stop_bot()
        time.sleep(2)
        start_bot()
    elif command == "status":
        check_status()
    elif command == "install":
        install_task()
    elif command == "uninstall":
        remove_task()
    else:
        print(f"Unknown command: {command}")
        print(__doc__)


if __name__ == "__main__":
    main()
